#include <cstdlib>
#include <cmath>
using namespace std; 
double integrate(double (*f)(double),double a,double b,int m=10)
    {   
    double h=(b-a)/2/m;   
    double s1=0;  
    for(int k=1;k<=m-1;k++)
        {
        s1+=4*f(a+(2*k-1)*h)+2*f(a+2*k*h);   
        }   
        s1+=f(a)+f(b)+4*f(a+(2*m-1)*h);   
        s1*=h/3;      
        return s1; 
        } 
        double F(double x)
        {  
            return 5*pow(x,4) - 3*pow(x,2) + 3; 
        }
